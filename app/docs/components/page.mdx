## Custom Components

You can create and use your own React components in MDX:

```tsx
// components/MyComponent.tsx
interface MyComponentProps {
  children: React.ReactNode
}

export function MyComponent({ children }: MyComponentProps) {
  return (
    <div className="my-custom-component">
      {children}
    </div>
  )
}
```

Then use it in your MDX:

```mdx
import { MyComponent } from '../components/MyComponent'

<MyComponent>
  This is my custom component content!
</MyComponent>
```
